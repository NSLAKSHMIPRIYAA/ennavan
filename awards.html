window.onload = function () { const flashLayer = document.getElementById("flashLayer"); const cameraSound = document.getElementById("cameraSound"); let running = true; function flashOnce() { // 📸 flash flashLayer.style.opacity = "1"; setTimeout(() => { flashLayer.style.opacity = "0"; }, 70); // shorter = sharper camera flash // 🔊 FIX: force crisp short click behavior try { cameraSound.pause(); // stop any overlap cameraSound.currentTime = 0; cameraSound.volume = 0.7; cameraSound.play().catch(() => {}); } catch (e) {} } // 🎬 LONGER + MORE REALISTIC BURST SEQUENCE let totalDuration = 5000; // 5 seconds of paparazzi chaos let start = Date.now(); function burstLoop() { if (!running) return; let now = Date.now(); if (now - start > totalDuration) { running = false; return; } flashOnce(); // 🔥 irregular timing = REAL camera crowd feel let delay = 80 + Math.random() * 120; setTimeout(burstLoop, delay); } burstLoop(); };
